Создать скрипт ownersort.sh, который в заданной папке копирует файлы в директории, названные по имени владельца каждого файла. 
Файл должен принадлежать соответствующему владельцу.
=====================================================

root@karartjom-Idea:/home/GB/Bashers# cat ownersort.sh
#!/bin/bash

dir=$(ls -l | tr -s ' ' '\t' | cut -f '3' | sort -u)
for i in $dir; do
        mkdir -p $i
done

dirfile=$(ls -l | tr -s ' ' '\t' | cut -f '3 9')
count=0
for i in $dirfile; do
        count=$((count+1))
        if (($count%2))
                then
                        dir=$i
                        echo $dir = $i zero
                else
                        if [ -f ./$i ]
                                then
                                        cp ./$i ./$dir/$i
                        fi
        fi
done
