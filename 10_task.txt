У меня подключение настроено через WiFi, поэтому тут уже есть созданные файлы с конфигурацией, разберем что у нас в файле с подключением к WiFi:
root@karartjom-idea:/etc/netplan# nano 01-network-manager-all.yml
network:
  version: 2
  wifis:
    NM-d0a07974-631d-45b2-8f5c-572dc30db7d2:
      renderer: NetworkManager
      match:
        name: "wlp9s0"
      dhcp4: no
      addresses: [192.168.1.22/24]
      routes:
        - to: default
          via: 192.168.1.1
      nameservers:
        addresses:
          - 8.8.8.8
          - 8.8.4.4
      dhcp6: true
      access-points:
        "Zorrki":
          auth:
            key-management: "psk"
            password: "**********"
          networkmanager:
            uuid: "d0a07974-631d-45b2-8f5c-572dc30db7d2"
            name: "Zorr"
            passthrough:
              wifi-security.auth-alg: "open"
              ipv6.addr-gen-mode: "default"
              ipv6.ip6-privacy: "-1"
              proxy._: ""
      networkmanager:
        uuid: "d0a07974-631d-45b2-8f5c-572dc30db7d2"
        name: "Zorrki"
